170    async def _authenticate(self, websocket: ClientConnection):
171        req = AgentRequest(auth_request=AgentRequestAuthRequest(token=self._token))
172        try:
173            await websocket.send(bytes(req))
174            # Fishjam will close the socket if auth fails and send a response on success
175            await websocket.recv(decode=False)
176        except ConnectionClosed:
177            raise AgentAuthError(websocket.close_reason or "")

 95        """
 96        Returns an infinite async iterator over the incoming messages from Fishjam to

 98        """
 99        while message := await self._ws.recv(decode=False):

106    async def add_track(self, options: AudioTrackOptions):
107        """
108        Adds a track to the connected agent, with the specified options and metadata.
109
110        Returns an instance of :class:`OutgoingTrack`, which can be used to send data
111        over the added track.
112        """
113        track_id = uuid.uuid4().hex
114        metadata_json = json.dumps(options.metadata)
115        message = AgentRequest(
116            add_track=AgentRequestAddTrack(
117                track=Track(
118                    id=track_id,
119                    type=TrackType.TRACK_TYPE_AUDIO,
120                    metadata=metadata_json,
121                ),
122                codec_params=AgentRequestAddTrackCodecParameters(
123                    encoding=options.encoding,
124                    sample_rate=options.sample_rate,
125                    channels=options.channels,
126                ),
127            )
128        )
129        await self._send(message)
130        return OutgoingTrack(id=track_id, session=self, options=options)
