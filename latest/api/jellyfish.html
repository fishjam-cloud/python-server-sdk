Jellyfish Python Server SDK

Python server SDK for the Jellyfish Media Server.

Read the docs here

Installation

pip install jellyfish-server-sdk

Usage

The SDK exports two main classes for interacting with Jellyfish server: RoomApi and Notifier.

RoomApi wraps http REST api calls, while Notifier is responsible for receiving real-time updates from the server.

RoomApi

Create a RoomApi instance, providing the jellyfish server address and api token

from jellyfish import RoomApi

room_api = RoomApi(server_address='localhost:5002', server_api_token='development')

You can use it to interact with Jellyfish, manage rooms, peers and components

# Create a room
jellyfish_address, room = room_api.create_room(video_codec='h264')
# 'localhost:5002', Room(components=[], config=RoomConfig(max_peers=None, video_codec='h264'), id='5a099a31-0eb2-4c28-84af-a1ec55c228af', peers=[]))

# Add peer to the room
from jellyfish import PeerOptionsWebRTC

peer_token, peer_webrtc = room_api.add_peer(room.id, options=PeerOptionsWebRTC())
# 'AgDYfrCSigFiAA', Peer(id='2869fb5', status=<PeerStatus.DISCONNECTED: 'disconnected'>, type='webrtc')

# Add component to the room
from jellyfish import ComponentOptionsHLS

component_hls = room_api.add_component(room.id, options=ComponentOptionsHLS())
# Component(actual_instance=ComponentHLS(id='c0dfab50-cafd-438d-985e-7b8f97ae55e3', metadata=ComponentMetadataHLS(low_latency=False, playable=False), type='hls'))

Notifier

Create Notifier instance

from jellyfish import Notifier

notifier = Notifier(server_address='localhost:5002', server_api_token='development')

Then define handlers for incoming messages

@notifier.on_server_notification
def handle_notification(server_notification):
    print(f'Received a notification: {server_notification}')

@notifier.on_metrics
def handle_metrics(metrics_report):
    print(f'Received WebRTC metrics: {metrics_report}')

After that you can start the notifier

async def test_notifier():
    notifier_task = asyncio.create_task(notifier.connect())

    # Wait for notifier to be ready to receive messages
    await notifier.wait_ready()

    # Create a room to trigger a server notification
    room_api = RoomApi()
    room_api.create_room()

    await notifier_task

asyncio.run(test_notifier())

# Received a notification: ServerMessageRoomCreated(room_id='69a3fd1a-6a4d-47bc-ae54-0c72b0d05e29')
# Received WebRTC metrics: ServerMessageMetricsReport(metrics='{}')

 42        self._server_address = f"{protocol}://{server_address}/socket/server/websocket"

 52    def on_server_notification(self, handler: Callable[[Any], None]):
 53        """
 54        Decorator used for defining handler for ServerNotifications
 55        i.e. all messages other than `ServerMessageMetricsReport`.
 56        """
 57        self._notification_handler = handler
 58        return handler
 59
 60    def on_metrics(self, handler: Callable[[ServerMessageMetricsReport], None]):
 61        """
 62        Decorator used for defining handler for `ServerMessageMetricsReport`.
 63        """
 64        self._metrics_handler = handler
 65        return handler
 66
 67    async def connect(self):
 68        """
 69        A coroutine which connects Notifier to Jellyfish and listens for all incoming
 70        messages from the Jellyfish.
 71
 72        It runs until the connection isn't closed.
 73
 74        The incoming messages are handled by the functions defined using the
 75        `on_server_notification` and `on_metrics` decorators.
 76
 77        The handlers have to be defined before calling `connect`,
 78        otherwise the messages won't be received.
 79        """
 80        async with client.connect(self._server_address) as websocket:
 81            try:
 82                self._websocket = websocket
 83                await self._authenticate()
 84
 85                if self._notification_handler:
 86                    await self._subscribe_event(
 87                        event=ServerMessageEventType.EVENT_TYPE_SERVER_NOTIFICATION
 88                    )
 89
 90                if self._metrics_handler:
 91                    await self._subscribe_event(
 92                        event=ServerMessageEventType.EVENT_TYPE_METRICS
 93                    )
 94
 95                self._ready = True
 96                if self._ready_event:
 97                    self._ready_event.set()
 98
 99                await self._receive_loop()
100            finally:
101                self._websocket = None
102
103    async def wait_ready(self) -> True:
104        """
105        Waits until the notifier is connected and authenticated to Jellyfish.
106
107        If already connected, returns `True` immediately.
108        """
109        if self._ready:
110            return True
111
112        if self._ready_event is None:
113            self._ready_event = asyncio.Event()
114
115        await self._ready_event.wait()
116
117    async def _authenticate(self):
118        msg = ServerMessage(
119            auth_request=ServerMessageAuthRequest(token=self._server_api_token)
120        )
121        await self._websocket.send(bytes(msg))
122
123        try:
124            message = await self._websocket.recv()
125        except ConnectionClosed as exception:
126            if "invalid token" in str(exception):
127                raise RuntimeError("Invalid server_api_token") from exception
128            raise
129
130        message = ServerMessage().parse(message)
131
132        _type, message = betterproto.which_one_of(message, "content")
133        assert isinstance(message, ServerMessageAuthenticated)
134
135    async def _receive_loop(self):
136        while True:
137            message = await self._websocket.recv()
138            message = ServerMessage().parse(message)
139            _which, message = betterproto.which_one_of(message, "content")
140
141            if isinstance(message, ServerMessageMetricsReport):
142                self._metrics_handler(message)
143            else:
144                self._notification_handler(message)
145
146    async def _subscribe_event(self, event: ServerMessageEventType):
147        request = ServerMessage(subscribe_request=ServerMessageSubscribeRequest(event))
148
149        await self._websocket.send(bytes(request))
150        message = await self._websocket.recv()
151        message = ServerMessage().parse(message)
152        _which, message = betterproto.which_one_of(message, "content")
153        assert isinstance(message, ServerMessageSubscribeResponse)
